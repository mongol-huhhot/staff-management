## DBアクセスCOMBOBOX

```code
{
  key: "department_id",
  label: "部署",
  type: "combobox",
  component: "AutoCombobox",

  valueField: "department_id",
  displayField: "department_name",

  dataSource: {
    type: "db",
    sqlCode: "get_department_list",
    params: {
      company_id: "company_id"
    }
  },

  props: {
    clearable: true,
    multiple: false,
    chips: false
  },

  rules: [
    { type: "required", message: "部署を選択してください" }
  ]
},


<script setup>
import { ref, computed, onMounted, watch } from 'vue'
import { useDbStore } from '@/stores/useDbStore'

const props = defineProps({
  modelValue: [String, Number, Object, Array, null],
  field: {
    type: Object,
    required: true
  },
  form: {
    type: Object,
    default: () => ({})
  }
})

const emit = defineEmits(['update:modelValue'])

const dbStore = useDbStore()

const items = ref([])
const loading = ref(false)

const value = computed({
  get: () => props.modelValue,
  set: v => emit('update:modelValue', v)
})

const itemTitle = computed(() => props.field.displayField || 'title')
const itemValue = computed(() => props.field.valueField || 'value')

function buildParams() {
  const params = {}

  const sourceParams = props.field.dataSource?.params || {}

  for (const [key, formKey] of Object.entries(sourceParams)) {
    params[key] = props.form[formKey]
  }

  return params
}

async function loadItems() {
  if (props.field.items) {
    items.value = props.field.items
    return
  }

  if (!props.field.dataSource) return

  loading.value = true

  try {
    const params = buildParams()

    const result = await dbStore.select({
      sqlCode: props.field.dataSource.sqlCode,
      params
    })

    items.value = result || []
  } catch (e) {
    console.error('AutoCombobox loadItems error:', e)
    items.value = []
  } finally {
    loading.value = false
  }
}

onMounted(() => {
  loadItems()
})

watch(
  () => props.field.dataSource?.params
    ? Object.values(buildParams())
    : [],
  () => {
    loadItems()
  }
)
</script>

<template>
  <v-combobox
    v-model="value"
    :label="field.label"
    :items="items"
    :item-title="itemTitle"
    :item-value="itemValue"
    :loading="loading"
    :clearable="field.props?.clearable ?? true"
    :multiple="field.props?.multiple ?? false"
    :chips="field.props?.chips ?? false"
    :return-object="field.props?.returnObject ?? false"
    :rules="field._rules || []"
  />
</template>

```
